Simple Component with closing tag

import React from 'react'

export default function SimpleNestComp() {
  return (
    <div>
      <Apple>hello</Apple>
    </div>
  )
}

function Apple(p) {
  return (<button>{p.children}</button>)
}

Nested Components with 2 children

import React from 'react'

export default function Testing2() {

        <p>hello</p>

Function as children

import React from 'react'

export default function FunctionAsChild() {
  function myfunc() {
    console.log('working')
  }
  return (
    <div>
      <Repeat numTimes={10}>
        hello world
        {myfunc}
      </Repeat>
    </div>
  )
}

function Repeat(props) {
  props.children[1]()
  
  // in case of single child
  // props.children() 
  
  return <b>{props.children[0]}</b>
}
