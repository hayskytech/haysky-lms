Props

props can be string, number, array, object, functionfs etc

Add this in component: Newbar.js

export default function Newbar(props) {
	return (
		<div>Hello {props.title} {props.name}</div>
	)
}

In Main component App.js

<Newbar name="Raju" title="Hello"/>

PropTypes

import PropTypes from 'prop-types'

add this after return at the bottom of component Newbar.js

Newbar.propTypes = {
	name: PropTypes.string,
	title: PropTypes.string
};

Default Props

Newbar.defaultProps = {
	name: 'Stranger',
	title: "Mr."
};

Different ways to pass the props

import React from 'react'
export default function App() {
  const todo = { fname: 'raj', lname: 'kumar' }
  const person = { man: 'ravi', age: 'kumar' }
  return (
    <div>
      {/* simple and best method */}
      <Child todo={todo} person={person} />

      {/* these two are same */}
      <Child {...todo} {...person} />
      <Child {...{ ...todo, ...person }} />

      {/* below both are same */}
      <Child p={{ todo, person }} />
      <Child p={{ todo: todo, person: person }} />

    </div>
  )
}
function Child(p) {
  return (
    <div>
      {JSON.stringify(p)}
    </div>
  )
}

Another way to recieve props

import React from 'react'
export default function App() {
  return (
    <div>
      <Child fname="apple" lname="bat" />
    </div>
  )
}
function Child({ fname, lname }) {
  return (
    <div>
      {fname}
      <br />
      {lname}
    </div>
  )
}
