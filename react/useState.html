States

Define a state and event handlers:

import { useState } from 'react'

Add these inside export default function

const [text, setText] = useState('Enter text')
const handleUpClick = () => {
	let newText = text.toUpperCase();
	setText(newText)
}
const handleOnChange = (event) => {
	setText(event.target.value)
}

this is the HTML part

<textarea className="form-control" id="myform" rows="3" value={text} onChange={handleOnChange}>

prev

for repeated statements, prev is compulsory.

const [count, setCount] = useState(0)
function increaseCounter(){
  setCount(prev => prev+1)
  setCount(prev => prev+1)
  setCount(prev => prev+1)
  setCount(count+1)
}

const [person, setPerson] = useState({name : 'John', age : 30})
function handleOnChange(e){
	setPerson({
    ...person, 
    name : e.target.value
  })
}

instead of ...person you can use ...prev

function handleOnChange(e){
  setPerson((prev) => ({
    ...prev, 
    name : e.target.value
  }))
}
