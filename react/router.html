Browser Router

function App() {
  return (<>
    <BrowserRouter>
      <Routes>
        <Route path="/" element={<MainMenu />}>
          <Route index element={<CodeGen />} />
          <Route path="about" element={<About />} />
          <Route path="contact" element={<Contact />} />
          <Route path="profile/:userid" element={<Profile />} />
          <Route path="*" element={<NoPage />} />
        </Route>
      </Routes>
    </BrowserRouter>
  </>)
}
