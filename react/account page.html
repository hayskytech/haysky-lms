Example Settings Page

import React, { useContext } from 'react'
import { List } from 'semantic-ui-react'
import { MyContext } from '../App'
import { Link } from 'react-router-dom'
import { doLogout } from '../firebaseConfig'

export default function Account() {
  const { theme, user, setTheme } = useContext(MyContext)
  return (
    <div>
      <h3>Account Settings</h3>

      <List divided relaxed="very" inverted={theme}>
        <List.Item as={Link} to="/courses">
          <List.Icon name='book' size='large' verticalAlign='middle' />
          <List.Content>
            <List.Header>Courses</List.Header>
            <List.Description>Add - Edit - Delete Courses</List.Description>
          </List.Content>
        </List.Item>
        <List.Item as={Link} to="/settings">
          <List.Icon name='setting' size='large' verticalAlign='middle' />
          <List.Content>
            <List.Header>Settings</List.Header>
            <List.Description>Change app settings</List.Description>
          </List.Content>
        </List.Item>
        <List.Item as={Link} to="/student_fields">
          <List.Icon name='user' size='large' verticalAlign='middle' />
          <List.Content>
            <List.Header>Student Fields</List.Header>
            <List.Description>Change student fields</List.Description>
          </List.Content>
        </List.Item>
        <List.Item as={Link} onClick={() => setTheme(!theme)}>
          <List.Icon name={theme ? 'moon' : 'sun'} size='large' verticalAlign='middle' />
          <List.Content>
            <List.Header>Change Theme</List.Header>
            <List.Description>Current: {theme ? 'Dark' : 'Light'}</List.Description>
          </List.Content>
        </List.Item>
        <List.Item as={Link} onClick={doLogout}>
          <List.Icon name='log out' size='large' color='red' verticalAlign='middle' />
          <List.Content>
            <List.Header>Logout</List.Header>
            <List.Description>{user?.email}</List.Description>
          </List.Content>
        </List.Item>

      </List>
    </div>
  )
}
