Sample AJAX request to get data from API

<p>Getting data from API</p>
<script>
	const url = "https://api.sampleapis.com/countries/countries"
	x = new XMLHttpRequest();
	x.onload = function () {
		data = JSON.parse(this.responseText)
		for (i = 0; i < data.length; i++) {
			// loop using data[i]
		}
		pre = document.createElement('pre');
		pre.textContent = JSON.stringify(data, undefined, 2)
		console.log(typeof (data))
		document.body.append(pre)
	}
	x.open("GET", url, true);
	x.send();
</script>
<style>
	pre {
		overflow-x: auto;
	}
</style>
